public void cargarTablaAutomatica() {
    // 1. Consulta que devuelve una lista de mapas
    List<Map<String, Object>> lista = session.createQuery("select new map(id as ID, nombre as Nombre, email as Email) from Cliente", Map.class).list();

    if (lista.isEmpty()) return;

    // 2. Extraer nombres de columnas automáticamente del primer registro
    Object[] columnas = lista.get(0).keySet().toArray();

    // 3. Crear el modelo con esas columnas
    DefaultTableModel dtm = new DefaultTableModel(columnas, 0);

    // 4. Llenar las filas dinámicamente
    for (Map<String, Object> fila : lista) {
        Object[] renglon = new Object[columnas.length];
        for (int i = 0; i < columnas.length; i++) {
            renglon[i] = fila.get(columnas[i]);
        }
        dtm.addRow(renglon);
    }

    // 5. Mostrar en la interfaz
    vista.tablaClientes.setModel(dtm);
}
